name: ci-lint
on:
  pull_request:
    branches: [ main ]
  push:
    branches: [ main ]
jobs:
  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with: { python-version: "3.11" }
      - run: pip install ruff==0.6.9
      - run: ruff check .
